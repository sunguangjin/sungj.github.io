    this.pausedAt = 0;

      // console.log('------------callback: play');

        // console.log('------------callback: stop');
        this.callback('stop', this);
      }
    }
  }
  // æš‚åœ
  pause() {
    const elapsed = this.context.currentTime - this.startedAt;
    this.stop();

    }
  }
